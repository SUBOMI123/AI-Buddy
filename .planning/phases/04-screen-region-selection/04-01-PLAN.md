```rust
/// Show the full-screen region-select overlay window.
/// Positions to cover the primary monitor exactly before showing.
/// Sets focus so Escape key events are received by the WebView. (D-10)
#[tauri::command]
pub async fn cmd_open_region_select(app: AppHandle) -> Result<(), String> {
    let win = app
        .get_webview_window("region-select")
        .ok_or_else(|| "region-select window not found".to_string())?;

    // Size to cover primary monitor exactly (Pitfall 3: never use fixed pixel dims)
    if let Ok(Some(monitor)) = win.primary_monitor() {
        let pos = monitor.position();
        let size = monitor.size();
        let _ = win.set_position(tauri::Position::Physical(
            tauri::PhysicalPosition::new(pos.x, pos.y),
        ));
        let _ = win.set_size(tauri::Size::Physical(
            tauri::PhysicalSize::new(size.width, size.height),
        ));
    }

    win.show().map_err(|e| e.to_string())?;
    win.set_focus().map_err(|e| e.to_string())?; // Pitfall 2: must set focus
    Ok(())
}

/// Hide the region-select overlay window without destroying it. (Pitfall 5)
#[tauri::command]
pub async fn cmd_close_region_select(app: AppHandle) -> Result<(), String> {
    let win = app
        .get_webview_window("region-select")
        .ok_or_else(|| "region-select window not found".to_string())?;
    win.hide().map_err(|e| e.to_string())
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| frontend → Rust (capture_region) | Frontend passes x, y, width, height as IPC arguments — untrusted input from WebView |
| region-select window → overlay window | Tauri global event carries region coordinates — cross-window IPC |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-01 | Tampering | `capture_region` coordinate arguments | mitigate | Validate x >= 0, y >= 0, width > 0, height > 0, and x+width <= monitor.width, y+height <= monitor.height before calling xcap. Returns descriptive Err strings. |
| T-04-02 | Information Disclosure | `capture_region` result (base64 JPEG) | accept | Same privacy model as Phase 2: capture is on-demand, not stored, user explicitly drew the region. No additional risk beyond full-screenshot baseline. |
| T-04-03 | Denial of Service | Very large region request | mitigate | Monitor bounds validation (T-04-01) caps region to physical monitor size. 1280px resize cap prevents oversized JPEG encoding. |
| T-04-04 | Elevation of Privilege | region-select window appears on top of all other apps | accept | alwaysOnTop is required for core mechanic. User explicitly initiates region selection. macOSPrivateApi:true already granted in Phase 1 for overlay. |
</threat_model>
