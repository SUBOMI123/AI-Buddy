            "min-height": "32px",
            opacity: props.disabled ? "0.4" : "1",
          }}
          onMouseEnter={(e) => {
            if (!props.disabled) {
              (e.currentTarget as HTMLButtonElement).style.color = "var(--color-text-primary)";
            }
          }}
          onMouseLeave={(e) => {
            (e.currentTarget as HTMLButtonElement).style.color = "var(--color-text-secondary)";
          }}
        >
          Try another way
        </button>
      );
    }
