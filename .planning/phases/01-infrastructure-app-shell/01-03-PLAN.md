**2. Create `src/lib/tauri.ts`** — IPC wrappers for Tauri commands:

```typescript
import { invoke } from "@tauri-apps/api/core";
import { listen } from "@tauri-apps/api/event";

export async function checkScreenPermission(): Promise<boolean> {
  return invoke<boolean>("check_screen_permission");
}

export async function requestScreenPermission(): Promise<boolean> {
  return invoke<boolean>("request_screen_permission");
}

export async function toggleOverlay(): Promise<void> {
  return invoke("cmd_toggle_overlay");
}

export function onOverlayShown(callback: () => void) {
  return listen("overlay-shown", callback);
}

export function onOverlayHidden(callback: () => void) {
  return listen("overlay-hidden", callback);
}
```

**8. Create `src/components/SidebarShell.tsx`** (per D-01, D-02):

The main container component that orchestrates all others.

Structure:
```
<div class="sidebar-shell">  <!-- 300px, full height, background var(--color-surface), border on inner edge -->
  <DragHandle />
  <div class="sidebar-content">  <!-- flex: 1, overflow-y: auto, padding var(--space-md) horizontal -->
    {/* Conditional rendering: */}
    {needsPermission ? <PermissionDialog /> : null}
    {!needsPermission && steps.length === 0 ? <EmptyState /> : null}
    {!needsPermission && permissionDenied ? <NoPermissionState /> : null}
    {!needsPermission && steps.length > 0 ? <GuidanceList steps={steps} /> : null}
  </div>
  <div class="sidebar-input-area">  <!-- padding var(--space-md), border-top var(--color-border) -->
    <TextInput onSubmit={handleSubmit} disabled={needsPermission} ref={setInputRef} />
  </div>
</div>
```

State management (using SolidJS `createSignal`):
- `needsPermission`: boolean — starts as `true`, check via `checkScreenPermission()` on mount
- `permissionDenied`: boolean — true if user clicked "Not Now"
- `steps`: string[] — guidance steps to display
- `inputRef`: HTMLInputElement reference for auto-focus

On mount (`onMount`):
1. Call `checkScreenPermission()`. If returns true, set `needsPermission = false`.
2. Listen for `overlay-shown` event via `onOverlayShown()` — when fired, focus the input ref (per D-08).

`handleSubmit(text)`:
- For Phase 1, set `steps` to `["Not connected yet -- AI guidance will be available in the next update."]` (per UI-SPEC stub response).

Sidebar CSS:

- Slide-in animation: `translateX(100%)` to `translateX(0)`, 200ms, `cubic-bezier(0.16, 1, 0.3, 1)` (per UI-SPEC animation)
