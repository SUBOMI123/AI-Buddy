```rust
fn play_cue_bytes(wav_bytes: &'static [u8]) {
    // Grab tokio handle so rodio's DeviceSink drop can find the runtime
    let rt_handle = tokio::runtime::Handle::try_current().ok();
    std::thread::spawn(move || {
        // Enter the runtime context so DeviceSink drop doesn't log "no runtime"
        let _guard = rt_handle.as_ref().map(|h| h.enter());
        use rodio::{DeviceSinkBuilder, Decoder};
        use std::io::Cursor;
        // rodio 0.22 API: DeviceSinkBuilder::open_default_sink() returns MixerDeviceSink.
        let Ok(sink) = DeviceSinkBuilder::open_default_sink() else {
            return; // No audio device — silently skip
        };
        let Ok(source) = Decoder::new(Cursor::new(wav_bytes)) else {
            return; // Decode error — silently skip
        };
        sink.mixer().add(source);
        // Hold the sink alive until playback finishes (~200ms ceiling for ptt cues)
        std::thread::sleep(std::time::Duration::from_millis(200));
    });
}
```

The key changes:
- `tokio::runtime::Handle::try_current().ok()` captures the handle from the caller's async context (before the thread is spawned).
- `let _guard = rt_handle.as_ref().map(|h| h.enter())` enters the runtime inside the thread so rodio's DeviceSink drop can resolve it.
- Renamed `handle` to `sink` for clarity (it is a MixerDeviceSink, not a generic handle).
- Removed the unused `duration_secs` binding.
